    email      VARCHAR(255)  NOT NULL DEFAULT '',

    created_at TIMESTAMPTZ   NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ   NOT NULL DEFAULT NOW()

CREATE TABLE public.clientes (
    id         SERIAL               PRIMARY KEY,
    first_name VARCHAR(255)         NOT NULL,
    last_name  VARCHAR(255)         NOT NULL,

    phone      VARCHAR(50)          NOT NULL CONSTRAINT chk_phone_format CHECK (phone ~ '^[+]?[0-9]{7,15}$'),
    status     public.cliente_status NOT NULL DEFAULT 'Activo',
    created_at TIMESTAMPTZ          NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ          NOT NULL DEFAULT NOW()
);

-- 4. INVENTARIO

    id           SERIAL                  PRIMARY KEY,
    cliente_id   INTEGER                 NOT NULL REFERENCES public.clientes(id) ON DELETE RESTRICT,
    name_article VARCHAR(255)            NOT NULL,
    description  VARCHAR(500),
    category     public.article_category NOT NULL DEFAULT 'Otro',
    amount       INTEGER                 NOT NULL DEFAULT 0 CHECK (amount >= 0),
    price        NUMERIC(12,2)           NOT NULL CHECK (price > 0),
    created_at   TIMESTAMPTZ             NOT NULL DEFAULT NOW(),
    updated_at   TIMESTAMPTZ             NOT NULL DEFAULT NOW()
);

-- 5. OPERACIONES DE EMPEÑO
CREATE TABLE public.pawns (
    id                SERIAL              PRIMARY KEY,
    profile_id        UUID                NOT NULL REFERENCES public.profile(id),
    article_id        INTEGER             NOT NULL REFERENCES public.articles(id),
    cliente_id        INTEGER             NOT NULL REFERENCES public.clientes(id),
    amount            INTEGER             NOT NULL CHECK (amount > 0),
    price             NUMERIC(12,2)       NOT NULL CHECK (price > 0),
    weight_grams      NUMERIC(8,2),
    installment_count INTEGER             NOT NULL DEFAULT 1 CHECK (installment_count >= 1),
    installments_paid INTEGER             NOT NULL DEFAULT 0 CHECK (installments_paid >= 0),
    installments_missed INTEGER           NOT NULL DEFAULT 0 CHECK (installments_missed >= 0),
    pawn_date         DATE                NOT NULL DEFAULT CURRENT_DATE,
    return_date       DATE                NOT NULL,
    status            public.pawn_status  NOT NULL DEFAULT 'Activo',
    notes             TEXT,
    updated_at        TIMESTAMPTZ         NOT NULL DEFAULT NOW(),
    CONSTRAINT chk_return_after_pawn  CHECK (return_date > pawn_date),
    CONSTRAINT chk_paid_lte_total     CHECK (installments_paid <= installment_count),
    CONSTRAINT chk_missed_lte_total   CHECK (installments_missed <= installment_count)
);

-- CORREGIDO: Columnas alineadas con PawnPaymentDao.java
-- - profile_id → created_by_profile_id (coincide con Java)
-- - amount_paid → amount (coincide con Java)
-- - CHECK >= 0 porque cuotas impagadas (is_missed=true) tienen amount=0
-- - Agregado: is_missed BOOLEAN para soportar HU-17

    id                    SERIAL        PRIMARY KEY,
    pawn_id               INTEGER       NOT NULL REFERENCES public.pawns(id) ON DELETE CASCADE,
    amount                NUMERIC(12,2) NOT NULL CHECK (amount >= 0),
    payment_date          DATE          NOT NULL DEFAULT CURRENT_DATE,
    notes                 VARCHAR(500),
    created_by_profile_id UUID          NOT NULL REFERENCES public.profile(id),
    is_missed             BOOLEAN       NOT NULL DEFAULT FALSE,
    created_at            TIMESTAMPTZ   NOT NULL DEFAULT NOW()
);

-- 8.1 Timestamp automático (updated_at)
CREATE OR REPLACE FUNCTION public.set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
    NEW.updated_at := NOW();
    RETURN NEW;
END;
$$;

CREATE TRIGGER trg_profile_upd BEFORE UPDATE ON public.profile FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_cliente_upd BEFORE UPDATE ON public.clientes FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_article_upd BEFORE UPDATE ON public.articles FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_pawn_upd    BEFORE UPDATE ON public.pawns FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();

-- 8.2 Registro automático de perfiles desde Supabase Auth
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
    INSERT INTO public.profile (id, email, full_name, rol)
    VALUES (
        NEW.id,
        NEW.email,
        COALESCE(NEW.raw_user_meta_data->>'full_name', 'Nuevo Usuario'),
        COALESCE((NEW.raw_user_meta_data->>'rol')::public.role_user, 'Empleado')
    );
    RETURN NEW;
END;
$$;

CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- 8.3 NUEVO: Validar peso obligatorio para Joyería en empeños (RF-04.13)
CREATE OR REPLACE FUNCTION public.validate_pawn_jewelry_weight()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE
    v_category public.article_category;
BEGIN
    SELECT category INTO v_category FROM public.articles WHERE id = NEW.article_id;

    IF v_category = 'Joyeria' AND (NEW.weight_grams IS NULL OR NEW.weight_grams <= 0) THEN
        RAISE EXCEPTION 'El peso en gramos es obligatorio para artículos de Joyería.'
            USING ERRCODE = 'CV002';
    END IF;

    RETURN NEW;
END;
$$;

CREATE TRIGGER trg_validate_pawn_weight
    BEFORE INSERT ON public.pawns
    FOR EACH ROW
    EXECUTE FUNCTION public.validate_pawn_jewelry_weight();

-- 8.4 NUEVO: Actualizar estado del empeño al registrar pago/cuota impagada (RF-04.10, RF-04.11, RF-04.12)
CREATE OR REPLACE FUNCTION public.update_pawn_status_on_payment()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE
    v_pawn RECORD;
BEGIN
    -- Bloqueo pesimista para evitar race conditions
    SELECT installment_count, installments_paid, installments_missed, status
    INTO v_pawn
    FROM public.pawns WHERE id = NEW.pawn_id FOR UPDATE;

    IF NEW.is_missed THEN
        -- Cuota impagada: incrementar missed (RF-04.12 / HU-17)
        UPDATE public.pawns
        SET installments_missed = installments_missed + 1
        WHERE id = NEW.pawn_id;

        -- Si supera 4 cuotas impagadas consecutivas → PERDIDO
        IF (v_pawn.installments_missed + 1) > 4 THEN
            UPDATE public.pawns SET status = 'Perdido' WHERE id = NEW.pawn_id;
            INSERT INTO public.audit_log (table_name, operation, record_id, new_data)
            VALUES ('pawns', 'UPDATE', NEW.pawn_id,
                    jsonb_build_object('auto_status', 'Perdido', 'reason', 'installments_missed > 4'));
        END IF;
    ELSE
        -- Pago real: incrementar paid, resetear missed a 0 (RF-04.10 / HU-16)
        UPDATE public.pawns
        SET installments_paid   = installments_paid + 1,
            installments_missed = 0
        WHERE id = NEW.pawn_id;

        -- Si paid alcanza count → FINALIZADO (RF-04.11)
        IF (v_pawn.installments_paid + 1) >= v_pawn.installment_count THEN
            UPDATE public.pawns SET status = 'Finalizado' WHERE id = NEW.pawn_id;
            INSERT INTO public.audit_log (table_name, operation, record_id, new_data)
            VALUES ('pawns', 'UPDATE', NEW.pawn_id,
                    jsonb_build_object('auto_status', 'Finalizado', 'reason', 'installments_paid = installment_count'));
        END IF;
    END IF;

    RETURN NEW;
END;
$$;

CREATE TRIGGER trg_pawn_payment_status
    AFTER INSERT ON public.pawn_payments
    FOR EACH ROW
    EXECUTE FUNCTION public.update_pawn_status_on_payment();

-- 9.1 Venta Atómica con Bloqueo de Stock
CREATE OR REPLACE FUNCTION public.register_sale(p_profile_id UUID, p_cliente_id INTEGER, p_items JSONB)
RETURNS INTEGER LANGUAGE plpgsql AS $$
DECLARE
    v_sale_id INTEGER;
    v_item    JSONB;
    v_stock   INTEGER;
BEGIN
    IF jsonb_array_length(p_items) = 0 THEN RAISE EXCEPTION 'No se puede registrar una venta sin artículos.'; END IF;

    INSERT INTO public.sales (profile_id, cliente_id) VALUES (p_profile_id, p_cliente_id) RETURNING id INTO v_sale_id;

    FOR v_item IN SELECT * FROM jsonb_array_elements(p_items) LOOP
        -- Bloqueo de fila para evitar condiciones de carrera (Race Conditions)
        SELECT amount INTO v_stock FROM public.articles WHERE id = (v_item->>'article_id')::INTEGER FOR UPDATE;
        
        IF v_stock < (v_item->>'amount')::INTEGER THEN
            RAISE EXCEPTION 'Stock insuficiente para el artículo ID %', (v_item->>'article_id') USING ERRCODE = 'CV001';
        END IF;

        INSERT INTO public.sales_details (sale_id, article_id, amount, unit_price)
        VALUES (v_sale_id, (v_item->>'article_id')::INTEGER, (v_item->>'amount')::INTEGER, (v_item->>'unit_price')::NUMERIC);

        UPDATE public.articles SET amount = amount - (v_item->>'amount')::INTEGER WHERE id = (v_item->>'article_id')::INTEGER;
    END LOOP;

    RETURN v_sale_id;
END;
$$;

-- 9.2 Expiración Masiva de Empeños
CREATE OR REPLACE FUNCTION public.fn_expire_overdue_pawns()
RETURNS INTEGER LANGUAGE plpgsql AS $$
DECLARE
    v_count INTEGER;
BEGIN
    UPDATE public.pawns
    SET    status     = 'Vencido',
           updated_at = NOW()
    WHERE  status      = 'Activo'
    AND    return_date < CURRENT_DATE;

    GET DIAGNOSTICS v_count = ROW_COUNT;

    IF v_count > 0 THEN
        INSERT INTO public.audit_log (table_name, operation, record_id, new_data)
        VALUES ('pawns', 'UPDATE', -1, jsonb_build_object('batch_expire_count', v_count, 'executed_at', NOW()));
    END IF;

    RETURN v_count;
END;
$$;

-- 9.3 NUEVO: Marcar empeño como Perdido manualmente (RF-04.12)
CREATE OR REPLACE FUNCTION public.fn_mark_pawn_lost(p_pawn_id INTEGER)
RETURNS VOID LANGUAGE plpgsql AS $$
BEGIN
    UPDATE public.pawns
    SET status = 'Perdido'
    WHERE id = p_pawn_id
      AND status IN ('Activo', 'Vencido');

    IF NOT FOUND THEN
        RAISE EXCEPTION 'El empeño % no está en estado válido para marcar como perdido.', p_pawn_id;
    END IF;

    INSERT INTO public.audit_log (table_name, operation, record_id, new_data)
    VALUES ('pawns', 'UPDATE', p_pawn_id,
            jsonb_build_object('manual_status', 'Perdido', 'executed_at', NOW()));
END;
$$;

-- 9.4 NUEVO: Generación thread-safe de número de factura (RF-08.2)
CREATE OR REPLACE FUNCTION public.next_invoice_number()
RETURNS TEXT LANGUAGE plpgsql AS $$
DECLARE
    v_today DATE := CURRENT_DATE;
    v_num   INTEGER;
BEGIN
    INSERT INTO public.invoice_sequence (seq_date, last_number)
    VALUES (v_today, 1)
    ON CONFLICT (seq_date) DO UPDATE SET last_number = invoice_sequence.last_number + 1
    RETURNING last_number INTO v_num;

    RETURN 'FACT-' || TO_CHAR(v_today, 'YYYYMMDD') || '-' || LPAD(v_num::TEXT, 4, '0');
END;
$$;

CREATE OR REPLACE VIEW public.v_dashboard AS
SELECT
    (SELECT COUNT(*) FROM public.pawns WHERE status = 'Activo')                                AS active_pawns,
    (SELECT COUNT(*) FROM public.pawns WHERE status = 'Vencido')                               AS overdue_pawns,
    (SELECT COUNT(*) FROM public.articles WHERE amount > 0)                                    AS total_article,
    (SELECT COUNT(*) FROM public.clientes WHERE status = 'Activo')                             AS total_clientes,
    COALESCE((SELECT SUM(price * amount) FROM public.pawns WHERE status = 'Activo'), 0)        AS total_active_value;
